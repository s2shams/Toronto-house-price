#!/bin/bash
# remove hyphens
read filename

x=1

while IFS=',' read -ra fields; do
    size=${fields[4]//-/}

    echo "${fields[0]},${fields[1]},${fields[2]},${fields[3]},$size,${fields[5]},${fields[6]},${fields[7]},${fields[8]}" >> output.csv
    ((x++))

    if [ $x -gt 10 ]; then
        break
    fi
done < $filename
